# /boot/firmware/config.txt — Cleaned & annotated for Arducam Quad-Camera HAT
# For full docs and more options see: http://rptl.io/configtxt
#

# Enable audio (loads snd_bcm2835)
dtparam=audio=on

# Enable DRM VC4 V3D driver with 512 MB CMA for hardware acceleration
dtoverlay=vc4-kms-v3d,cma-512

# Allow two framebuffers for multi-display setups
max_framebuffers=2

# Prevent firmware from creating an initial video= setting; use kernel’s default
disable_fw_kms_setup=1

# Run the kernel in 64-bit mode
arm_64bit=1

# Disable display overscan compensation
disable_overscan=1

# Boost CPU and GPU to maximum performance
arm_boost=1

# Automatically detect and load overlays for any connected DSI displays
display_auto_detect=1

# Automatically load initramfs files if found on boot
auto_initramfs=1

[cm4]
# Enable host mode on the built-in XHCI USB controller
# (remove this if you need legacy DWC2 or USB gadget mode)
otg_mode=1

[cm5]
# Enable DWC2 in host mode for USB on Compute Module 5
dtoverlay=dwc2,dr_mode=host

[all]
# ─── Arducam Quad-HAT config (DO NOT REMOVE OR REORDER) ────────────────────

# Disable stock Raspberry camera auto-detect so only our overlays run
camera_auto_detect=0

# Load the 64 MP sensor on CSI0 (must be first)
dtoverlay=arducam-64mp,cam0

# Load PiVariety driver for the Pi-camera modules on CSI0’s multiplexer
dtoverlay=arducam-pivariety,cam1
dtoverlay=arducam-pivariety,cam2
dtoverlay=arducam-pivariety,cam3

# ─── Required I²C channels for all four sensors ────────────────────────────

# GPIO-header I²C-1 (unused by hat but safe to leave on)
dtparam=i2c_arm=on

# VC-sideband I²C (camera reset / control bus)
dtparam=i2c_vc=on

# CSI0 sideband I²C (mux channel A)
dtparam=i2c_csi_dsi0=on

# CSI1 sideband I²C (mux channel B)
dtparam=i2c_csi_dsi1=on
